    document.querySelectorAll('.like-btn').forEach(button => {
        button.addEventListener('click', (e) => {
            const movieId = parseInt(e.target.getAttribute('data-id'));
            likeMovie(movieId);
        });
    });
   
    // Details buttons
    document.querySelectorAll('.details-btn').forEach(button => {
        button.addEventListener('click', (e) => {
            const movieId = parseInt(e.target.getAttribute('data-id'));
            showDetailsPopup(movieId);
        });
    });
}


// Toggle favorite status
function likeMovie(movieId) {
    const movieIndex = movies.findIndex(movie => movie.id === movieId);
    if (movieIndex !== -1) {
        movies[movieIndex].favorite = !movies[movieIndex].favorite;
        updateLocalStorage();
        renderAllMovies();
    }
}


// Show movie details in modal
function showDetailsPopup(id) {
    const movie = movies.find((movie) => movie.id === id);
    const modal = document.getElementById("movieModal");
    const modalBody = document.getElementById("modalBody");


    if (!movie) return;


    modalBody.innerHTML = `
        <div class="modal-header">
            <h2>${movie.title} <span class="modal-year">(${movie.year})</span></h2>
        </div>
        <div class="modal-main">
            <div class="modal-poster">
                <img src="${movie.poster}" alt="${movie.title}" />
            </div>
            <div class="modal-info">
                <p><strong>Director:</strong> ${movie.director}</p>
                <p><strong>Genre:</strong> ${movie.genre}</p>
                <p><strong>Status:</strong> ${movie.favorite ? '❤ Liked' : '🤍 Not Liked'}</p>
            </div>
        </div>
        <div class="modal-description">
            <h3>Description</h3>
            <p>${movie.description}</p>
        </div>
        <div class="modal-cast">
            <h3>Cast</h3>
            <ul>
                ${movie.actors.map(actor => `
                    <li>
                        <img src="${actor.image}" alt="${actor.actor_name}" />
                        <strong>${actor.actor_name}</strong> as <em>${actor.character_name}</em>
                    </li>
                `).join('')}
            </ul>
        </div>
    `;


    modal.classList.remove("hidden");
}


// Search movies by title
function searchMovies() {
    const searchTerm = searchInput.value.toLowerCase().trim();
   
    if (searchTerm === '') {
        renderAllMovies();
        return;
    }
   
    const filteredMovies = movies.filter(movie =>
        movie.title.toLowerCase().includes(searchTerm)
    );
   
    displayMovies(favoritesList, filteredMovies, `Search Results for "${searchTerm}"`);
}


// Filter movies by category
function filterByCategory(category) {
    const filteredMovies = movies.filter(movie =>
        movie.genre.toLowerCase().includes(category.toLowerCase())
    );
   
    displayMovies(favoritesList, filteredMovies, `${category} Movies`);
}


// Render all movies (both favorites and regular)
function renderAllMovies() {
    displayMovies(favoritesList, movies);
    displayMovies(favoritesList1, movies.filter(movie => movie.favorite), "Favorite Movies");
}

// Event listeners for search
searchButton.addEventListener('click', searchMovies);
searchInput.addEventListener('keyup', (e) => {
    if (e.key === 'Enter') {
        searchMovies();
    }
});


// Event listeners for category filtering
categoryLinks.forEach(link => {
    link.addEventListener('click', (e) => {
        e.preventDefault();
        const category = link.textContent.replace(/^\w+\s/, ''); // Remove icon text
        filterByCategory(category);
    });
});


// Setup navbar functionality
function setupNavbar() {
    // Navbar active state
    const navLinks = document.querySelectorAll('.nav-links a');
    navLinks.forEach(link => {
        link.addEventListener('click', function() {
            navLinks.forEach(lnk => lnk.classList.remove('active'));
            this.classList.add('active');
        });
    });
   
    // Close dropdown when clicking outside
    document.addEventListener('click', function(event) {
        if (!event.target.matches('.dropbtn')) {
            const dropdowns = document.getElementsByClassName('dropdown-content');
            for (let i = 0; i < dropdowns.length; i++) {
                const openDropdown = dropdowns[i];
                if (openDropdown.style.display === 'block') {
                    openDropdown.style.display = 'none';
                }
            }
        }
    });
   
    // Toggle dropdown menu
    document.querySelector('.dropbtn').addEventListener('click', function(e) {
        e.stopPropagation();
        const dropdown = this.nextElementSibling;
        dropdown.style.display = dropdown.style.display === 'block' ? 'none' : 'block';
    });
   
    // Navbar scroll effect
    const navbar = document.querySelector('.navbar');
    const heroSection = document.querySelector('.hero-section');
   
    window.addEventListener('scroll', function() {
        if (window.scrollY > 10) {
            navbar.classList.add('scrolled');
        } else {
            navbar.classList.remove('scrolled');
        }
    });
   
    // Adjust hero section height
    function adjustHeroHeight() {
        const navbarHeight = navbar.offsetHeight;
        heroSection.style.paddingTop = `${navbarHeight}px`;
    }
   
    adjustHeroHeight();
    window.addEventListener('resize', adjustHeroHeight);
}
